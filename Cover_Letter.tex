% Rho Class - Cover Letter Template
% --------------------------------------------------------
% 使用说明：
% 1. 使用 class/cover_letter 文档类
% 2. 填写发件人/收件人信息
% 3. 使用 \makeheader 和 \makeclosing 生成标准格式
% --------------------------------------------------------
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\documentclass{class/cover_letter}

%----------------------------------------------------------
% 发件人信息 (Sender)
%----------------------------------------------------------
\sendername{Prof. First Last}
\senderaffiliation{Department of Computer Science \\ University Name}
\senderaddress{123 Science Road, City, Country, 00000}
\senderemail{email@university.edu}

%----------------------------------------------------------
% 收件人信息 (Recipient)
%----------------------------------------------------------
\recipientname{Dr. Editor Name}
\recipienttitle{Editor-in-Chief}
\recipientjournal{Journal Name}
\recipientaddress{Editorial Office Address}

%----------------------------------------------------------
% 主题与称呼
%----------------------------------------------------------
\subject{Submission of Manuscript "Paper Title Here"}
\opening{Dear Dr. Editor,}
\closing{Sincerely,}

%----------------------------------------------------------
% 正文
%----------------------------------------------------------
\begin{document}

\makeheader

I am writing to submit our manuscript entitled \textbf{"Paper Title"} for consideration of publication in \textit{Journal Name}.

Detailed content goes here...

\begin{itemize}
    \item Contribution 1
    \item Contribution 2
\end{itemize}

We look forward to your response.

\makeclosing

\end{document}
